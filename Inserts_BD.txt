select v.nombre, count(p.nro_pedido)'cant_ventas', sum(dp.precio)
from Vendedores v, Pedidos p,DetallesPedidos dp
where v.id_vendedor=p.id_vendedor and p.nro_pedido=dp.nro_pedido
group by v.nombre
order by 'cant_ventas' desc


select count(dp.cantidad)*100/(select count(dp.cantidad) from DetallesPedidos dp)
from Pedidos p, Remeras r, Tipo t, DetallesPedidos dp
where dp.nro_pedido=p.nro_pedido and dp.id_remera=r.id_remera and r.id_tipo=t.id_tipo and t.id_tipo=1
